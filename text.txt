// Interfaces
interface IsPerson {
    name : string;
    age : number;
    speak(a: string) : void;
    spend(b: number) : number;
}

const me : IsPerson = {
    name : 'Tung',
    age: 30,
    speak(text: string) : void {
        console.log(text);
    },
    spend(amount: number) :number {
        console.log('i spend', amount);
        return amount;
    }
};

const greetPerson = (person : IsPerson) => {
    console.log('Hello', person.name);
}

greetPerson(me);

console.log(me);
